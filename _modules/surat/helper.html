[docs]def get_tendik():
    tgl = str(datetime.datetime.today().toordinal())
    seed = hashlib.sha256(str('alfarabi' + tgl).encode(encoding='utf-8')).hexdigest()
    url = 'https://sihrd.ums.ac.id/umar/v1/uniid/tendik?user=aan493&hash=' + seed
    r = requests.get(url, verify=False)
    tendik = r.json()
    list_tendik = list()
    for data in tendik['rows']:
        list_tendik.append((data['UniID'], data['Nama lengkap']))
    return list_tendik

[docs]def get_jabatan():
    url_ws = settings.SIHRD_URL + "umar/v2/jabatan"
    r = requests.get(url_ws, auth=(settings.SIHRD_WS_UNAME, settings.SIHRD_WS_PWD))
    list_jabatan = list()
    for data in r.json()['rows']:
        if data['penjabat']:
            list_jabatan.append((data['kode_jabatan'], data['nama']+' '+data['unit']))
    return list_jabatan
