    else:
        ajuan = get_object_or_404(AjuanInsentifTBI, pk=id)

    if request.method == 'POST':
        form = AccTBIForm(request.POST, instance=ajuan)
        if form.is_valid():
            request.session['ntbi'] -= 1
            request.session['ninsentif'] -= 1
            tbi = form.save(commit=False)

            if 'ajukan' in request.POST:
                if in_kaprodi_group(request.user) | in_dekan_group(request.user):
                    tbi.approval_kaprodi = True
                    tbi.tgl_approval_kaprodi = datetime.date.today()
                    tbi.status = 'DIPROSES'
                elif in_lppi_group(request.user):
                    tbi.approval_lppi = True
                    tbi.tgl_approval_lppi = datetime.date.today()
                elif in_wr1_group(request.user):
                    tbi.approval_wr1 = True
                    tbi.tgl_approval_wr1 = datetime.date.today()
                    tbi.status = 'DISETUJUI'
                elif in_wr2_group(request.user):
                    tbi.approval_wr2 = True
                    tbi.tgl_approval_wr2 = datetime.date.today()
            elif 'revisi' in request.POST:
                if in_kaprodi_group(request.user) | in_dekan_group(request.user):
                    tbi.status = 'REVISI'
                elif in_lppi_group(request.user):
                    #tbi.approval_kaprodi = False
                    #tbi.tgl_approval_kaprodi = None
                    tbi.status = 'REVISI'
                elif in_wr1_group(request.user):
                    tbi.approval_lppi = False
                    tbi.tgl_approval_lppi = None
                    tbi.status = 'REVISI'
            elif 'tolak' in request.POST:
                tbi.status = 'DITOLAK'
            tbi.save()
        return redirect('approval_tbi')

    data = AjuanInsentifTBI.objects.get(id=id)
    form = AccTBIForm(instance=ajuan)
    sk_terbaru = SKInsentif.objects.order_by('-tgl_sk')[0]
    form.fields['insentif'].queryset = Insentif.objects.filter(masih_berlaku = True, jenis__sk__id = sk_terbaru.id, jenis__kode = 'TBI' )
    return render(request, 'insentif/acc_tbi.html',{'admininsentif':'TBI','form':form,'data':data})
